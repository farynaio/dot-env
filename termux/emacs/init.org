    (use-package dired
      :straight nil
       :custom
      (dired-kill-when-opening-new-dired-buffer t)
      (dired-dwim-target t)
      :config
      (setq dired-listing-switches "-ah1")
      (add-hook 'dired-mode-hook #'dired-hide-details-mode)
      (define-key dired-mode-map (kbd "<backspace>") #'dired-up-directory)
      (define-key dired-mode-map (kbd "DEL") #'dired-up-directory) ; optional
      ;; allow dired-find-alternate-file (it's disabled by default)
      (put 'dired-find-alternate-file 'disabled nil)
      ;; RET: replace current dired buffer with target (no new window)
      (define-key dired-mode-map (kbd "RET") #'dired-find-alternate-file)
      ;; mouse-1: same behavior when clicking

      ;; (define-key dired-mode-map [mouse-1] #'dired-find-alternate-file)

      (define-key dired-mode-map [mouse-1] nil)

  ;; When clicking a file in dired, just move point there instead of visiting it.
  (defun my/dired-click-move-point (event)
    "Move point to the clicked line in dired without opening the file."
    (interactive "e")
    (let* ((posn (event-start event))
           (win  (posn-window posn))
           (buf  (window-buffer win))
           (pos  (posn-point posn)))
      (with-selected-window win
        (with-current-buffer buf
          (goto-char pos)
          (dired-move-to-filename)))))

  ;; ;; Bind left mouse click in dired to the do-nothing mover
  (define-key dired-mode-map [mouse-2] #'my/dired-click-move-point))

#+begin_src emacs-lisp :tangle init.el :eval never-export
  ;; Kill shell buffer when shell exits
  (use-package shell
    :straight nil
    :hook (shell-mode . my-shell-mode-hook-func)
    :config
    (defun my-shell-mode-hook-func ()
      (set-process-sentinel (get-buffer-process (current-buffer))
                            'my-shell-mode-kill-buffer-on-exit))
    (defun my-shell-mode-kill-buffer-on-exit (process state)
      (message "%s" state)
      (if (or
           (string-match "exited abnormally with code.*" state)
           (string-match "finished" state))
          (kill-buffer (current-buffer))))

    ;; default shell buffer
    (setq explicit-shell-file-name "/data/data/com.termux/files/usr/bin/fish")
    ;; use shell-file-name for subprocesses
    (setq shell-file-name explicit-shell-file-name)

    )
